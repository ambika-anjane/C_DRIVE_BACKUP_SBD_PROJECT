CREATE TABLE  "DIM_INFOR". DIM_PROMOTIONS 
   (	PROMO_KEY NUMERIC(6,0), 
	PROMO_CODE VARCHAR(20), 
	PROMO_NAME VARCHAR(30), 
	PROMO_SUBCATEGORY VARCHAR(30), 
	PROMO_CATEGORY VARCHAR(30), 
	PROMO_COST NUMERIC(10,2), 
	PROMO_BEGIN_DATE DATE, 
	PROMO_END_DATE DATE, 
	PROMO_TOTAL VARCHAR(15), 
	INSERT_DT DATE, 
	LAST_UPDATE_DT DATE, 
	DW_INSERT_DT DATE, 
	DW_UPDATE_DT DATE, 
	MD5_CHECKSUM VARCHAR(32), 
	CDC_FLAG VARCHAR(1)
   );
